Conexión Supabase (Base de Datos en la Nube)

Conectado a Supabase correctamente.
Ver SQL para crear las tablas en Supabase 
-- Ejecutar en el SQL Editor de Supabase

create table if not exists justificantes (
  id text primary key, folio text, alumno text, no_control text, grado text, grupo text,
  fecha_ausencia text, fecha_expedicion text, tutor text, telefono_tutor text,
  motivo text, observaciones text, validador text,
  created_at timestamptz default now()
);

create table if not exists permisos (
  id text primary key, folio text, alumno text, no_control text, grado text, grupo text,
  fecha text, hora text, hora_regreso text, persona text, parentesco text,
  telefono text, autoriza text, validador text, motivo text,
  created_at timestamptz default now()
);

create table if not exists citatorios (
  id text primary key, folio text, alumno text, no_control text, grado text, grupo text,
  tutor text, telefono_tutor text, fecha_cita text, hora_cita text,
  emite text, validador text, estado text, motivo text, acuerdos text,
  fecha_creacion text, created_at timestamptz default now()
);

create table if not exists reportes (
  id text primary key, folio text, alumno text, no_control text, grado text, grupo text,
  fecha text, tipo_falta text, reporta text, descripcion text,
  medida text, seguimiento text, created_at timestamptz default now()
);

create table if not exists bitacora (
  id text primary key, folio text, tipo text, gravedad text, fecha text,
  hora text, lugar text, reporta text, involucrados text, descripcion text,
  acciones text, seguimiento text, created_at timestamptz default now()
);

create table if not exists alumnos (
  id text primary key, no_control text unique, nombre text,
  apellido_paterno text, apellido_materno text,
  grado text, grupo text, turno text, curp text,
  tutor text, telefono_tutor text, created_at timestamptz default now()
);
